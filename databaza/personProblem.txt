@startuml

entity Person {
    * id : serial
    --
    * name : varchar
    * age : int
    * gender : varchar
    * mail : varchar
    premium : boolean
}

entity Problem {
    * id : serial
    --
    severity : varchar
}

entity AffectedBodyPart {
    * id : serial
    --
    * name : varchar
}

entity MuscleGroup {
    * id : serial
    --
    * name : varchar
    muscles : [Muscle]
}

entity Muscle {
    * id : serial
    --
    * name : varchar
    spasmus : boolean
    stretchedOut : boolean
    abbreviated : boolean
    stiffness : boolean
}

Person ||..o{ Problem
Problem }|..|{ AffectedBodyPart
AffectedBodyPart ||..|{ Muscle
AffectedBodyPart ||..|{ MuscleGroup
MuscleGroup ||..|{ Muscle

@enduml
